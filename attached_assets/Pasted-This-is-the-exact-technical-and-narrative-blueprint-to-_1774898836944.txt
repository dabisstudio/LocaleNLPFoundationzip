**`tailwind.config.js`**

```javascript

module.exports = {
  content: ["./app/**/*.{js,ts,jsx,tsx}", "./components/**/*.{js,ts,jsx,tsx}"],
  theme: {
    extend: {
      colors: {
        // The Infinite Canvas
        background: "#04040A", // Deep Midnight Indigo (Not pure black)

          primary: "#FAFAFA",
```

#### B. The Interactive Bento Box Card (Mouse Spotlight Effect)
This is the hallmark of modern, high-end tech sites. As the user moves their mouse over the card, a subtle radial gradient follows the cursor, illuminating the border and background.

```tsx
'use client';
import { useRef, useState } from 'react';

export default function TechCard({ title, description, metric }) {
  const divRef = useRef<HTMLDivElement>(null);
  const [isFocused, setIsFocused] = useState(false);
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const [opacity, setOpacity] = useState(0);

  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
    if (!divRef.current || isFocused) return;
    const div = divRef.current;
    const rect = div.getBoundingClientRect();
    setPosition({ x: e.clientX - rect.left, y: e.clientY - rect.top });
  };

  return (
    <div
      ref={divRef}
      onMouseMove={handleMouseMove}
      onMouseEnter={() => setOpacity(1)}
      onMouseLeave={() => setOpacity(0)}
      className="relative w-full overflow-hidden rounded-xl border border-white/10 bg-surface p-8 transition-colors hover:bg-surfaceHover"
    >
      {/* The Mouse Spotlight */}
      <div
        className="pointer-events-none absolute -inset-px opacity-0 transition duration-300"
        style={{
          opacity,
          background: `radial-gradient(600px circle at ${position.x}px ${position.y}px, rgba(245,166,35,.1), transparent 40%)`,
        }}
      />
      
      <div className="relative z-10 flex flex-col h-full">
        <div className="font-mono text-xs text-accent-cyan mb-4">{metric}</div>
        <h3 className="font-display text-2xl text-white mb-3">{title}</h3>
        <p className="text-text-secondary leading-relaxed">{description}</p>
      </div>
    </div>
  );
}
```
